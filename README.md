  # BOOK TABLE 
  ```sql 
 CREATE TABLE BOOK_TABLE
(
BOOK_ID FOREIGN KEY,
BOOK_NAME VARCHAR(200),
BOOK_SERIAL_NUMBER NUMBER AUTO GENERATED BY DEFAULT AS IDENTITY (START WITH 1000 INCREMENT BY 11),
BOOK_AVAILABLITY_STATUS VARCHAR(10)
);
 ```
